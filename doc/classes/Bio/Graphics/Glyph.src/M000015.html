  def self.circle(args) 
    args = { 
      :radius => 10, 
      :fill_color => 'red', 
      :stroke => "black", 
      :stroke_width => 1, 
      :style => ""}.merge!(args)
      args[:x_center] = args[:x]
      args[:y_center] = args[:y]
      [:x, :y].each {|e| args.delete(e)}
    [Bio::Graphics::Primitive.new(:circle, args)]
  end
