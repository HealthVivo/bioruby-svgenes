  def self.span(args)
    args = {
      :height => 10,
      :fill_color => 'red',
      :stroke => "black",
      :stroke_width => 1,
      :style => "fill-opacity:1;"
    }.merge!(args)
    args[:x1] = args[:x]
    args[:x2] = args[:x] + args[:width]
    args[:y1] = args[:y]
    args[:y2] = args[:y]
    [Bio::Graphics::Primitive.new(:line, args)]
  end
