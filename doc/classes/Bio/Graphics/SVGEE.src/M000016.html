  def initialize(args)
    opts = {:width => '100%', :height => '100%'}.merge!(args)
    @width = opts[:width]
    @height = opts[:height]
    @style= opts[:style]
    @primitives = []
    @defs = []
    @supported_primitives = [:circle, :rectangle, :ellipse, :line, :polyline, :text]
  end
